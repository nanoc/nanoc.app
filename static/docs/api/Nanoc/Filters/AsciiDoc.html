def run(content, params = {})
  # Run command
  stdout = ''
  stderr = ''
  status = systemu(
    %w( asciidoc -o - - ),
    'stdin'  => content,
    'stdout' => stdout,
    'stderr' => stderr)

  # Show errors
  unless status.success?
    $stderr.puts stderr
    raise "AsciiDoc filter failed with status #{status}"
  end

  # Get result
  stdout
end
