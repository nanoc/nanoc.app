Overview

def [](*args)
  if 1 == args.size && args.first.is_a?(String)
    self.item_with_identifier(args.first)
  else
    @items[*args]
  end
end

def at(arg)
  if arg.is_a?(String)
    self.item_with_identifier(arg)
  else
    @items[arg]
  end
end
