def run
  time_before = Time.now

  load_site
  check_for_deprecated_usage

  puts 'Compiling siteâ€¦'
  run_listeners_while do
    site.compile
    prune
  end

  time_after = Time.now
  puts
  puts "Site compiled in #{format('%.2f', time_after - time_before)}s."
end
