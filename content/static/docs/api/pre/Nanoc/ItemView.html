def fetch(key, fallback=NONE, &block)
  res = @item[key] # necessary for dependency tracking

  if @item.attributes.key?(key)
    res
  else
    if !fallback.equal?(NONE)
      fallback
    elsif block_given?
      yield(key)
    else
      raise KeyError, "key not found: #{key.inspect}"
    end
  end
end
